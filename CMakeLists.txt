cmake_minimum_required(VERSION 2.8.3)
project(autoliv)

## Compile as C++11, supported in ROS Kinetic and newer
# add_compile_options(-std=c++11)
add_definitions(-std=c++11)

## Find catkin macros and libraries
## if COMPONENTS list like find_package(catkin REQUIRED COMPONENTS xyz)
## is used, also find other catkin packages
find_package(catkin REQUIRED COMPONENTS
  roscpp
  rospy
  std_msgs
  message_generation
  dataspeed_can_msgs
)


## Generate messages in the 'msg' folder
add_message_files(
   FILES
   RawPolarShort.msg
   RawPolarLong.msg
   TargetPolarShort.msg
   TargetPolarLong.msg
   TargetCartesian.msg
   TargetCartesianMid.msg
   TargetCartesianMul.msg
   FreespaceSegments.msg
   TargetCartesianLong.msg
)


## Generate added messages and services with any dependencies listed here
generate_messages(
   DEPENDENCIES
   std_msgs
)


catkin_package(
  LIBRARIES ${PROJECT_NAME}
)

include_directories(
  ${catkin_INCLUDE_DIRS}
)

add_executable( autoliv_node src/node.cpp src/AutolivNode.cpp)
add_dependencies(autoliv_node ${${PROJECT_NAME}_EXPORTED_TARGETS} ${catkin_EXPORTED_TARGETS})
target_link_libraries( autoliv_node ${catkin_LIBRARIES})

install(TARGETS
    autoliv_node
    DESTINATION ${CATKIN_PACKAGE_SHARE_DESTINATION}
)

install(DIRECTORY launch
    DESTINATION ${CATKIN_PACKAGE_SHARE_DESTINATION}
)
